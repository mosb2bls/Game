Texture2D menuTexture : register(t0);
SamplerState samplerState : register(s0);

struct PSIn
{
    float4 pos   : SV_POSITION;
    float2 uv    : TEXCOORD0;
    float3 tint  : TEXCOORD1;
    float alpha  : TEXCOORD2;
};

float4 PS(PSIn i) : SV_TARGET
{
    float4 texColor = menuTexture.Sample(samplerState, i.uv);
    return float4(texColor.rgb * i.tint, texColor.a * i.alpha);
}
