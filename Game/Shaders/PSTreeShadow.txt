struct PSIn
{
    float4 pos    : SV_POSITION;
    float3 normal : NORMAL;
    float2 uv     : TEXCOORD0;
};

float4 PS(PSIn i) : SV_TARGET
{
    // Circular shadow with soft edges
    float2 center = float2(0.5, 0.5);
    float dist = length(i.uv - center) * 2.0;
    
    // Soft falloff
    float shadow = 1.0 - smoothstep(0.0, 1.0, dist);
    
    // Dark shadow color with alpha
    float opacity = 0.65 * shadow; 
    
    return float4(0.0, 0.0, 0.0, opacity);
}
